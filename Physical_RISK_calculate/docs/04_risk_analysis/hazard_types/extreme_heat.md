```python
    ê·¼ê±°:

    """

        lat=lat, lon=lon, method='nearest'
```

```python
    duration_norm = min(future_max_dur / 30.0, 1.0)

    hwd_norm = min(hwd_ratio / 3.0, 1.0)

```

```python
def calculate_impervious_ratio(building_info, land_cover_raster):

    lon = building_info['longitude']
```
